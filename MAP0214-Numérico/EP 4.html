def euler(t, y, z, tStop, h):
    while t < tStop:
        tabela1.add_row((t, y, z)) # A cada passo, a função preenche a tabela.
        z = z + h*g(t,y,z)
        t = t + h
        y = y + h*z
    erro_relativo_y = abs((y - y_exact(t))/y_exact(t))
    erro_relativo_z = abs((z - z_exact(t))/z_exact(t))
    print("A solução para t = %.2f é y(%.2f) = %.4f e z(%.2f) = y'(%.2f) = %.4f"%(tabela1['t'][-1], tabela1['t'][-1], tabela1['y'][-1], tabela1['t'][-1], tabela1['t'][-1], tabela1['z'][-1]))
    print("\n O erro relativo para y é err_y = %.4f e para z é err_z = %.4f"%(erro_relativo_y, erro_relativo_z))
    return tabela1

def RK4(t, y, z, tStop, h):
    while(t <= tStop):
        tabela2.add_row((t, y, z)) # A cada passo, a função preenche a tabela.
        k_1y = h*z
        k_1z = h*g(t,y,z)
        
        k_2y = h*(z + k_1z/2)
        k_2z = h*g(t + h/2, y + k_1y/2,z + k_1z/2)
        
        k_3y = h*(z + k_2z/2)
        k_3z = h*g(t + h/2, y + k_2y/2,z + k_2z/2)
        
        k_4y = h*(z + k_3z)
        k_4z = h*g(t + h, y + k_3y,z + k_3z)
        
        t = t + h
        y = y + (k_1y + 2*k_2y + 2*k_3y + k_4y)/6
        z = z + (k_1z + 2*k_2z + 2*k_3z + k_4z)/6
    
    erro_relativo_y = abs((y - y_exact(t))/y_exact(t))
    erro_relativo_z = abs((z - z_exact(t))/z_exact(t))
    print("A solução para t = %.2f é y(%.2f) = %.6f e z(%.2f) = y'(%.2f) = %.6f"%(tabela2['t'][-1], tabela2['t'][-1], tabela2['y'][-1], tabela2['t'][-1], tabela2['t'][-1], tabela2['z'][-1]))
    print("\n O erro relativo para y é err_y = %.6f e para z é err_z = %.6f"%(erro_relativo_y, erro_relativo_z))
    return tabela2

def RK4(t, x, v, npassos, h, g):
    T=[]
    X=[]
    V=[]
    for i in range(npassos):
        T.append(t)
        X.append(x)
        V.append(v)
        k_1x = h*v
        k_1v = h*g(t,x,v)
        
        k_2x = h*(v + k_1v/2)
        k_2v = h*g(t + h/2, x + k_1x/2,v + k_1v/2)
        
        k_3x = h*(v + k_2v/2)
        k_3v = h*g(t + h/2, x + k_2x/2,v + k_2v/2)
        
        k_4x = h*(v + k_3v)
        k_4v = h*g(t + h, x + k_3x,v + k_3v)
        
        t = t + h
        x = x + (k_1x + 2*k_2x + 2*k_3x + k_4x)/6
        v = v + (k_1v + 2*k_2v + 2*k_3v + k_4v)/6
    return T, X, V

xplot=[]
fplot=[]
for F in tqdm(np.arange(0., 0.7, 0.0005)):
    t=0
    x=-1
    v=1
    w=1
    h=0.01*2*np.pi/w
    g = lambda t, x, v: 1/2 * x * (1-x**2) - 0.25*v + F*np.cos(w*t)
    T_transiente, X_transiente, V_transiente = RK4(t, x, v, 200000, h, g) # RK4 retorna os vetores que carregam a evolucao do sistema até 200.000 passos,

    t=T_transiente[-1]
    x=X_transiente[-1]
    v=V_transiente[-1]
    for i in range(100):
        T, X, V = RK4(t, x, v, 1000, h, g)
        t = T[-1]
        x = X[-1]
        v = V[-1]
