tabela1 = Table(names=('Iterao N', 'a', 'b', 'xm', 'f(xm)', ), dtype=('i4', 'f4', 'f4','f4', 'f4')) # Formato da minha tabela 1

def raiz_Bissec(funcao, x, precisao):
    a = x[0]
    b = x[-1]
    N=0
    while(abs(b-a) > precisao): # Enquanto no atinjo a preciso, o mtodo continua
        xm = (a+b)/2
        fm = funcao(xm)
        tabela1.add_row((N, a, b, xm, fm)) # A cada passo, a funo preenche a tabela.
        if (funcao(a)*funcao(xm)>0): # Se os sinais forem iguais, xa recebe xm
            a = xm
            fm = funcao(a)
        else:
            b = xm
            fm = funcao(b)
        N = N+1
    x0 = b #Aps atingir minha preciso, o valor final  um dos estremos.
    
    print('O zero da funo est no intervalo de x: [%.3f, %.3f]'%(a, b))
    return x0

def funcao1(x):
    return x**3 - np.cos(x**2)

def derivada_funcao1(x):
    return 3 * x**2 + np.sin(x**2) * 2*x
        
def raiz_Newton(funcao, x, precisao):
    x0 = x
    for i in range(100): # Aqui, defino o mximo de iteraes como 100.
        if abs(funcao(x0)) < precisao: # x0 prximo o suficiente de ser uma raiz
            print ('Raiz encontrada aps %.i iteraes' %i)
            tabela2.add_row((i, x0, funcao(x0), derivada_funcao1(x0)))
            return x0
        tabela2.add_row((i, x0, funcao(x0), derivada_funcao1(x0))) # A cada passo, a funo preenche a tabela.
        x0 = x0 - funcao(x0)/derivada_funcao1(x0)

def derivada_secante(f,x0,x1): # Substituir a derivada do mtodo de Newton pelo retorno dessa funo , resulta no mtodo das secantes.
    return (f(x1) - f(x0))/(x1-x0)

def raiz_Secantes(funcao, x0, x1, precisao):
    for N in range(100): # Aqui, defino o mximo de iteraes como 100.
        if abs(funcao(x0)) < precisao: # x0 prximo o suficiente de ser uma raiz
            tabela3.add_row((N, x0, funcao(x0), x1, funcao(x1))) # Preenche o ltimo resultado na tabela
            return x0
        tabela3.add_row((N, x0, funcao(x0), x1, funcao(x1))) # A cada passo, a funo preenche a tabela.
        aux = x1
        x1 = x1 - funcao(x1)/derivada_secante(funcao, x0, x1)
        x0 = aux
